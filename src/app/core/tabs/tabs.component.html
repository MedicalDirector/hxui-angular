Tabs

The tab component is a small complete navigation component. Do I really need to explain what tabs are? :)

Usage

Add the TabsModule to AppModule or SharedModule.

    
      import {{ '{' }}TabsModule} from "@hxui-angular/packages/tabs/tabs.module";

      @NgModule({{ '{' }}
        imports: [TabsModule.forRoot(),...]
      })
      export class AppModule(){{ '{' }}
    
    
    <hx-tabset>
      <hx-tab heading='Tab 1 heading'>Tab 1 content</hx-tab>
      <hx-tab>
        <template tabHeading>Tab 2 heading</template>
        Tab 2 content
      </hx-tab>
    </hx-tabset>

      import {{ '{' }} Component, OnInit } from '@angular/core';

      @Component({{ '{' }}
        selector: 'app-tabs',
        templateUrl: './tabs.component.html'
      })
      export class TabsComponent implements OnInit {{ '{' }}

        public tabs: any[] = [
          {{ '{' }}title: 'Dynamic Title 1', content: 'Dynamic content 1'},
          {{ '{' }}title: 'Dynamic Title 2', content: 'Dynamic content 2', disabled: true},
          {{ '{' }}title: 'Dynamic Title 3', content: 'Dynamic content 3', removable: true},
          {{ '{' }}title: 'Dynamic Title 4', content: 'Dynamic content 4', customClass: 'customClass'}
        ];

        public alertMe(): void {{ '{' }}
          setTimeout(function (): void {{ '{' }}
            alert('You\'ve selected the alert tab!');
          });
        }

        public setActiveTab(index: number): void {{ '{' }}
          this.tabs[index].active = true;
        }

        public removeTabHandler(): void {{ '{' }}
          console.log('Remove Tab handler');
        }

        constructor() {{ '{' }}

        ngOnInit() {{ '{' }}
        }
      }

API Reference

TabsetComponent
Inputs Type Description
justified boolean if true tabs fill the container and have a consistent width
type string navigation context class: 'tabs' or 'pills'
vertical boolean if true tabs will be placed vertically
hasInfo boolean if true info tab classes will be added
TabDirective
Inputs Type Description
active boolean tab active state toggle
customClass string if set, will be added to the tab's class atribute
disabled boolean if true tab can not be activated
heading string tab header text
id string tab id
removable boolean if true tab can be removable, additional button will appear
Outputs Description
deselect fired when tab became inactive, $event:Tab equals to deselected instance of Tab component
removed fired before tab will be removed, $event:Tab equals to instance of removed tab
select fired when tab became active, $event:Tab equals to selected instance of Tab component
TabsetConfig
Properties Type Description
type string provides default navigation context class: 'tabs' or 'pills'
